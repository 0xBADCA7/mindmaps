1 mindmaps.StatusBarView = function() {
  2 	var self = this;
  3 	var $statusbar = $("#statusbar");
  4 
  5 	this.init = function() {
  6 	};
  7 
  8 	this.createButton = function(id, text) {
  9 		return $("<button/>", {
 10 			id : "statusbar-button-" + id
 11 		}).button({
 12 			label : text
 13 		}).click(function() {
 14 			if (self.buttonClicked) {
 15 				self.buttonClicked(id);
 16 			}
 17 		}).prependTo($statusbar.find(".buttons"));
 18 	};
 19 
 20 	this.getContent = function() {
 21 		return $statusbar;
 22 	};
 23 };

 25 mindmaps.StatusBarPresenter = function(eventBus, view) {
 26 	var buttonCounter = 0;
 27 	var buttonIdPanelMap = {};
 28 	var statusController = new mindmaps.StatusNotificationController(eventBus,
 29 			view.getContent());
 30 
 31 	view.buttonClicked = function(id) {
 32 		buttonIdPanelMap[id].toggle();
 33 	};
 34 
 35 	this.go = function() {
 36 		view.init();
 37 
 38 	};
 39 
 40 	this.addEntry = function(panel) {
 41 		var id = buttonCounter++;
 42 		var $button = view.createButton(id, panel.caption);
 43 		panel.setHideTarget($button);
 44 		buttonIdPanelMap[id] = panel;
 45 	};
 46 };

 48 mindmaps.StatusNotificationController = function(eventBus, view) {
 49 	var $anchor = $("<div class='notification-anchor'/>").css({
 50 		position : "absolute",
 51 		right : 20
 52 	}).appendTo(view);
 53 
 54 	eventBus.subscribe(mindmaps.Event.DOCUMENT_SAVED, function() {
 55 		var n = new mindmaps.Notification($anchor, {
 56 			position : "topRight",
 57 			expires : 2500,
 58 			content : "Mind map saved"
 59 		});
 60 	});
 61 };
