1 mindmaps.ToolBarView = function() {
  2 	var self = this;
  3 
  4 	this.init = function() {
  5 	};
  6 
  7 	this.addButton = function(button, alignFunc) {
  8 		// var $button = this.createButton(button);
  9 		alignFunc(button.asJquery());
 10 	};
 11 
 12 	this.addButtonGroup = function(buttons, alignFunc) {
 13 		var $buttonset = $("<span/>");
 14 		buttons.forEach(function(button) {
 15 			// var $button = self.createButton(button);
 16 			$buttonset.append(button.asJquery());
 17 		});
 18 		$buttonset.buttonset();
 19 		alignFunc($buttonset);
 20 	};
 21 
 22 	this.addMenu = function(menu) {
 23 		this.alignRight(menu.getContent());
 24 	};
 25 
 26 	this.alignLeft = function($button) {
 27 		$button.appendTo("#toolbar .buttons-left");
 28 	};
 29 
 30 	this.alignRight = function($button) {
 31 		$button.appendTo("#toolbar .buttons-right");
 32 	};
 33 };

 35 /**
 36  * Toolbar button object
 37  * 
 38  * @constructor
 39  * @param {mindmaps.Command} command
 40  */
 41 
 42 mindmaps.ToolBarButton = function(command) {
 43 	this.command = command;
 44 
 45 	// callback to update display state
 46 	var self = this;
 47 	command.subscribe(mindmaps.Command.Event.ENABLED_CHANGED, function(enabled) {
 48 		if (self.setEnabled) {
 49 			self.setEnabled(enabled);
 50 		}
 51 	});
 52 };
 53 
 54 mindmaps.ToolBarButton.prototype.isEnabled = function() {
 55 	return this.command.enabled;
 56 };
 57 
 58 mindmaps.ToolBarButton.prototype.click = function() {
 59 	this.command.execute();
 60 };
 61 
 62 mindmaps.ToolBarButton.prototype.getTitle = function() {
 63 	return this.command.label;
 64 };
 65 
 66 mindmaps.ToolBarButton.prototype.getToolTip = function() {
 67 	var tooltip = this.command.description;
 68 
 69 	var shortcut = this.command.shortcut;
 70 	if (shortcut) {
 71 		tooltip += " [" + shortcut.toUpperCase() + "]";
 72 	}
 73 
 74 	return tooltip;
 75 };
 76 
 77 mindmaps.ToolBarButton.prototype.getId = function() {
 78 	return "button-" + this.command.id;
 79 };
 80 
 81 mindmaps.ToolBarButton.prototype.asJquery = function() {
 82 	var self = this;
 83 	var $button = $("<div/>", {
 84 		id : this.getId(),
 85 		title : this.getToolTip()
 86 	}).click(function() {
 87 		self.click();
 88 	}).button({
 89 		label : this.getTitle(),
 90 		disabled : !this.isEnabled()
 91 	});
 92 
 93 	var icon = this.command.icon;
 94 	if (icon) {
 95 		$button.button({
 96 			icons : {
 97 				primary : icon
 98 			}
 99 		});
100 	}
101 
102 	// callback to update display state
103 	this.setEnabled = function(enabled) {
104 		$button.button(enabled ? "enable" : "disable");
105 	};
106 
107 	return $button;
108 };

110 mindmaps.ToolBarMenu = function(title, icon) {
111 	var self = this;
112 	this.$menuWrapper = $("<span/>", {
113 		"class" : "menu-wrapper"
114 	});
115 
116 	this.$menuButton = $("<div/>").button({
117 		label : title,
118 		icons : {
119 			primary : icon,
120 			secondary : "ui-icon-triangle-1-s"
121 		}
122 	}).appendTo(this.$menuWrapper);
123 
124 	this.$menu = $("<div/>", {
125 		"class" : "menu"
126 	}).click(function() {
127 		/*
128 		 * hack to hide menu on click. visibility on hover is triggered by CSS.
129 		 * force display:none for a short time and remove class immediately
130 		 * again.
131 		 */
132 		self.$menu.addClass("hidden");
133 		setTimeout(function() {
134 			self.$menu.removeClass("hidden");
135 		}, 10);
136 	}).appendTo(this.$menuWrapper);
137 
138 	this.add = function(buttons) {
139 		if (!Array.isArray(buttons)) {
140 			buttons = [ buttons ];
141 		}
142 
143 		buttons.forEach(function(button) {
144 			var $button = button.asJquery().removeClass("ui-corner-all")
145 					.addClass("menu-item");
146 			this.$menu.append($button);
147 		}, this);
148 
149 		// last item gets rounded corners
150 		this.$menu.children().last().addClass("ui-corner-bottom").prev()
151 				.removeClass("ui-corner-bottom");
152 	};
153 
154 	this.getContent = function() {
155 		return this.$menuWrapper;
156 	};
157 };

159 mindmaps.ToolBarPresenter = function(eventBus, commandRegistry, view,
160 		mindmapModel) {
161 	// map commands to buttons
162 	function commandToButton(commandType) {
163 		var command = commandRegistry.get(commandType);
164 		return new mindmaps.ToolBarButton(command);
165 	}
166 
167 	function commandsToButtons(commands) {
168 		return commands.map(commandToButton);
169 	}
170 
171 	// populate toolbar
172 
173 	// node buttons
174 	var nodeCommands = [ mindmaps.CreateNodeCommand, mindmaps.DeleteNodeCommand ];
175 	var nodeButtons = commandsToButtons(nodeCommands);
176 	view.addButtonGroup(nodeButtons, view.alignLeft);
177 
178 	var undoCommands = [ mindmaps.UndoCommand, mindmaps.RedoCommand ];
179 	var undoButtons = commandsToButtons(undoCommands);
180 	view.addButtonGroup(undoButtons, view.alignLeft);
181 
182 	var clipboardCommands = [ mindmaps.CopyNodeCommand,
183 			mindmaps.CutNodeCommand, mindmaps.PasteNodeCommand ];
184 	var clipboardButtons = commandsToButtons(clipboardCommands);
185 	view.addButtonGroup(clipboardButtons, view.alignLeft);
186 
187 	// file menu
188 	var fileMenu = new mindmaps.ToolBarMenu("Mind map", "ui-icon-document");
189 	var fileCommands = [ mindmaps.NewDocumentCommand,
190 			mindmaps.OpenDocumentCommand, mindmaps.SaveDocumentCommand,
191 			mindmaps.CloseDocumentCommand ];
192 	var fileButtons = commandsToButtons(fileCommands);
193 	fileMenu.add(fileButtons);
194 
195 	view.addMenu(fileMenu);
196 
197 	view.addButton(commandToButton(mindmaps.HelpCommand), view.alignRight);
198 
199 	this.go = function() {
200 		view.init();
201 	};
202 };
